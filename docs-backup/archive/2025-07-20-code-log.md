### 2. Domain Model Integration

```python
# services/mcp/resources.py - Real TF-IDF scoring
def _calculate_relevance_score_with_domain_model(self, content: str, query: str) -> float:
    if self.content_extractor:
        relevance_score = self.content_extractor.calculate_relevance_score(content, query)
        return relevance_score.value  # Sophisticated TF-IDF
    else:
        return self._calculate_basic_relevance_score(content, query)  # Fallback
```
