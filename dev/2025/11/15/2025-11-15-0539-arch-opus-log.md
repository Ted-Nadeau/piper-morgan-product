## 6:00 AM - Skills MCP Design

### Proposed Architecture

```python
class SkillsMCP:
    """Meta-MCP that provides executable skills to agents"""

    skills = {
        "standup_to_issues": StandupToIssuesSkill(),
        "document_analysis": DocumentAnalysisSkill(),
        "multi_system_update": MultiSystemUpdateSkill(),
        "batch_operations": BatchOperationsSkill(),
    }

    def get_skill_definition(self, skill_name: str) -> str:
        """Return minimal definition for agent context"""
        return f"""
        Skill: {skill_name}
        Description: {self.skills[skill_name].description}
        Parameters: {self.skills[skill_name].parameters}
        Usage: await execute_skill('{skill_name}', params)
        """  # ~200 tokens per skill

    async def execute_skill(self, skill_name: str, params: dict):
        """Execute pre-written, tested code pattern"""
        skill = self.skills[skill_name]
        return await skill.execute(params)  # No tokens to model
```

### Example Skill Implementation

```python
class StandupToIssuesSkill:
    """Encapsulates standup → GitHub → Slack → Notion workflow"""

    async def execute(self, params: dict):
        # All data stays in execution environment
        standup_data = await self.parse_standup(params['standup_text'])

        issues_created = []
        for item in standup_data['action_items']:
            issue = await github.create_issue(
                title=item['title'],
                body=item['description'],
                labels=['standup', 'action-item']
            )
            issues_created.append(issue['number'])

        # Notify Slack
        await slack.post_message(
            channel='#standups',
            text=f"Created {len(issues_created)} issues from standup"
        )

        # Update Notion
        await notion.update_page(
            page_id=params['notion_page'],
            content={'issues': issues_created}
        )

        return {
            'success': True,
            'issues_created': len(issues_created),
            'summary': f"Processed standup: {len(issues_created)} issues"
        }  # Only summary returns to model (~50 tokens)
```
