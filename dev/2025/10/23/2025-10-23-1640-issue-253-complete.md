### Cost Calculation Algorithm

```python
# Per-token pricing calculation
prompt_cost = (prompt_tokens / 1000) * pricing["prompt"]
completion_cost = (completion_tokens / 1000) * pricing["completion"]
total_cost = prompt_cost + completion_cost

# Example: GPT-4 with 1000 prompt + 500 completion tokens
# Cost = (1000/1000 * $0.03) + (500/1000 * $0.06) = $0.06
```

### Database Schema (Ready for Implementation)

```sql
-- Usage tracking table
CREATE TABLE api_usage_logs (
    id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    provider VARCHAR(50) NOT NULL,
    model VARCHAR(100) NOT NULL,
    prompt_tokens INTEGER NOT NULL,
    completion_tokens INTEGER NOT NULL,
    total_tokens INTEGER NOT NULL,
    estimated_cost DECIMAL(10, 4) NOT NULL,
    conversation_id UUID,
    feature VARCHAR(50),
    created_at TIMESTAMP NOT NULL
);

-- Budget management table
CREATE TABLE user_budgets (
    id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    budget_type VARCHAR(20) NOT NULL,
    amount DECIMAL(10, 2) NOT NULL,
    provider VARCHAR(50),
    alert_thresholds DECIMAL(5, 2)[],
    created_at TIMESTAMP NOT NULL
);
```
