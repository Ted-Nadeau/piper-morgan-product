    def store_key(provider, api_key, username=None):
        key_name = f"{username}_{provider}_api_key" if username else f"{provider}_api_key"

    def get_key(provider, username=None):
        key_name = f"{username}_{provider}_api_key" if username else f"{provider}_api_key"
